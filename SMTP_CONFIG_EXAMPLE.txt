# SMTP настройки для отправки email
# Замените на ваши реальные данные

# Хост SMTP сервера (например, smtp.gmail.com, smtp.mail.ru, smtp.yandex.ru)
SMTP_HOST=smtp.gmail.com

# Порт SMTP сервера (обычно 587 для TLS или 465 для SSL)
SMTP_PORT=587

# Ваш email адрес для отправки
SMTP_USER=your-email@gmail.com

# Пароль приложения (для Gmail) или обычный пароль
SMTP_PASSWORD=your-app-password

# Email адрес клиники, куда будут приходить заявки
CLINIC_EMAIL=clinic@whitewolf.ru
